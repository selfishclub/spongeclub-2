```python
import mybrain

    def setUp(self):
        self.home = tempfile.mkdtemp()
```

```python
"""나를 알아가는 내 OS — 기록 엔진 (append + 반추 집계). 표준 라이브러리만 사용."""
import argparse
import json
import os
from datetime import datetime, timedelta

DEFAULT_HOME = os.path.join(os.path.expanduser("~"), "mybrain")
CATS = ("thought", "time", "input", "want", "mood", "misc")


def _parse_now(s):
    for fmt in ("%Y-%m-%dT%H:%M:%S", "%Y-%m-%dT%H:%M"):
        try:
            return datetime.strptime(s, fmt)
        except ValueError:
            continue
    raise SystemExit("bad --now (예: 2026-07-13T21:30:05): " + s)


def _make_id(now):
    return now.strftime("%Y%m%d-%H%M%S")


def _ensure_dirs(home):
    for sub in ("log", "daily", "weekly", "meta"):
        os.makedirs(os.path.join(home, sub), exist_ok=True)


def _log_path(home, ym):
    return os.path.join(home, "log", ym + ".jsonl")


def _update_tags(home, tags):
    if not tags:
        return
    path = os.path.join(home, "meta", "tags.md")
    existing = set()
    if os.path.exists(path):
        with open(path, encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line.startswith("- "):
                    existing.add(line[2:].strip())
    new = [t for t in tags if t not in existing]
    if new:
        with open(path, "a", encoding="utf-8") as f:
            if not existing:
                f.write("# 태그 사전\n\n")
            for t in new:
                f.write("- " + t + "\n")


def cmd_add(args):
    home = args.home
    _ensure_dirs(home)
    now = _parse_now(args.now)
    if args.cat not in CATS:
        raise SystemExit("cat must be one of " + ", ".join(CATS))
    occurred = args.occurred or now.strftime("%Y-%m-%d")
    occ_date = datetime.strptime(occurred, "%Y-%m-%d").date()
    tags = [t.strip() for t in (args.tags or "").split(",") if t.strip()]
    rec = {
        "id": args.correct or _make_id(now),
        "ts": now.strftime("%Y-%m-%dT%H:%M:%S"),
        "occurred": occurred,
        "cat": args.cat,
        "tags": tags,
        "text": args.text,
        "raw": args.raw if args.raw is not None else args.text,
    }
    if args.dur is not None:
        rec["dur_min"] = args.dur
    with open(_log_path(home, occ_date.strftime("%Y-%m")), "a", encoding="utf-8") as f:
        f.write(json.dumps(rec, ensure_ascii=False) + "\n")
    _update_tags(home, tags)
    print(json.dumps(rec, ensure_ascii=False))


def build_parser():
    p = argparse.ArgumentParser(prog="mybrain")
    sub = p.add_subparsers(dest="cmd", required=True)

    a = sub.add_parser("add")
    a.add_argument("--now", required=True)
    a.add_argument("--cat", required=True)
    a.add_argument("--tags", default="")
    a.add_argument("--text", required=True)
    a.add_argument("--raw", default=None)
    a.add_argument("--occurred", default=None)
    a.add_argument("--dur", type=int, default=None)
    a.add_argument("--correct", default=None)
    a.add_argument("--home", default=DEFAULT_HOME)
    a.set_defaults(func=cmd_add)
    return p


def main(argv=None):
    args = build_parser().parse_args(argv)
    args.func(args)


if __name__ == "__main__":
    main()
```

```python
def _load_records(home, start_date, end_date):
    # start_date~end_date가 걸치는 월 파일들을 읽어 id별 last-wins 병합 후
    # occurred가 범위 안인 레코드만 반환
    months = set()
    d = start_date
    while d <= end_date:
        months.add(d.strftime("%Y-%m"))
        d += timedelta(days=1)
    merged = {}
    for ym in sorted(months):
        path = _log_path(home, ym)
        if not os.path.exists(path):
            continue
        with open(path, encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                r = json.loads(line)
                merged[r["id"]] = r  # 나중 줄이 이김
    out = []
    for r in merged.values():
        try:
            od = datetime.strptime(r["occurred"], "%Y-%m-%d").date()
        except (KeyError, ValueError):
            continue
        if start_date <= od <= end_date:
            out.append(r)
    return out


def cmd_reflect(args):
    now = _parse_now(args.now)
    today = now.date()
    if args.range == "today":
        start = end = today
    elif args.range == "week":
        start, end = today - timedelta(days=6), today
    else:
        raise SystemExit("range must be today|week")
    recs = _load_records(args.home, start, end)
    by_cat, time_by_tag = {}, {}
    time_total, inputs_count = 0, 0
    wants, moods = [], []
    for r in recs:
        c = r.get("cat", "misc")
        by_cat[c] = by_cat.get(c, 0) + 1
        if c == "time":
            dur = r.get("dur_min", 0) or 0
            time_total += dur
            for t in r.get("tags", []):
                time_by_tag[t] = time_by_tag.get(t, 0) + dur
        elif c == "want":
            wants.append({"text": r.get("text", ""), "tags": r.get("tags", [])})
        elif c == "mood":
            moods.append(r.get("text", ""))
        elif c == "input":
            inputs_count += 1
    result = {
        "range": args.range,
        "from": start.isoformat(),
        "to": end.isoformat(),
        "total": len(recs),
        "by_cat": by_cat,
        "time": {"total_min": time_total, "by_tag": time_by_tag},
        "wants": wants,
        "moods": moods,
        "inputs_count": inputs_count,
    }
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

````markdown
# 나를 알아가는 내 OS (기록 + 반추)

텔레그램에 던진 걸 자동으로 쌓고, 물어보면 나를 되돌려주는 개인 OS.
기록·집계의 결정적 부분은 `mybrain.py`가, 판단(무슨 카테고리·태그인지)은 이 스킬이 맡는다.

> 스크립트 경로: `C:\Users\USER\.claude\skills\mybrain\mybrain.py`
> 데이터: `C:\Users\USER\mybrain\` (자동 생성, --home 안 넘기면 여기가 기본)

## 언제 쓰나
- 텔레그램 봇으로 개인 기록성 메시지가 도착했을 때 (생각/시간/링크/바람/기분).
- 사용자가 "오늘 어때", "이번 주 돌아봐", "반추해줘"라고 할 때.
- 봇이 매긴 분류를 사용자가 정정할 때.

## 캡처 — 메시지가 오면 (순서대로)

1. **명령/질문인지 먼저 판단.** "오늘 어때?" 같은 반추 요청이면 아래 [반추] 섹션으로. 그 외 일상 내용이면 기록으로 처리.
2. **카테고리 1개** 고르기:
   - 💭 `thought` 생각·고민·깨달음
   - ⏱️ `time` "뭘 하며 시간 썼다" (시간량 있으면 분으로)
   - 🔗 `input` 읽은 것·링크·영상·정보
   - 🎯 `want` 하고 싶다·되고 싶다·언젠가
   - 😌 `mood` 기분·에너지·컨디션
   - 📥 `misc` 애매하면 여기 (버리지 말 것)
3. **태그** 붙이기: 주제/영역(`오픽` `AI공부` `가족` `건강` `돈` 등). **`C:\Users\USER\mybrain\meta\tags.md`를 먼저 보고 이미 있는 태그를 재사용**한다(동의어 남발 금지). 없으면 새로 만들되 신중히.
4. **시간·날짜 추출:** "3시간"→`--dur 180`, "30분"→`--dur 30`. "어제/그저께"면 그 날짜를 `--occurred YYYY-MM-DD`로. 안 적혀 있으면 생략(오늘로 처리됨).
5. **엔진 호출** (지금 실제 시각을 `--now`에 ISO로):
   ```
   python "C:\Users\USER\.claude\skills\mybrain\mybrain.py" add \
     --now 2026-07-13T21:30:05 --cat time --tags 오픽,기획 --dur 180 \
     --text "오전에 오픽 컨텐츠 기획" --raw "오전에 오픽 컨텐츠 기획 3시간 했어"
   ```
   - `--text`는 정리된 핵심, `--raw`는 사용자 원문 그대로.
   - 출력 JSON의 `id`를 기억해 둔다(정정에 씀).
6. **확인 답장** (텔레그램 reply):
   ```
   ⏱️ 시간 · #오픽 #기획 · 180분으로 기록했어요
   ```
   이모지는 카테고리에 맞게(💭/⏱️/🔗/🎯/😌/📥).

## 정정 — 사용자가 "그거 X 아니야"

- 사용자가 분류를 고치면(예: "그거 시간 아니고 바람이야"), **직전 기록의 `id`로 다시 add**:
  ```
  python "...\mybrain.py" add --now <지금> --correct 20260713-213005 \
    --cat want --tags 오픽 --text "..." --raw "..."
  ```
- 같은 `id`로 한 줄 더 쌓이고, 반추는 마지막 것을 채택한다(원본은 안 지움).
- "방금 그거"인데 id가 애매하면, 오늘 로그의 마지막 줄 id를 쓴다.
````
